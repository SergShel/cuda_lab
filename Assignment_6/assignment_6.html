class ConvBlock(nn.Module):
    """
    Simple convolutional block: Conv + Norm + Act + Dropout
    """
    
    def __init__(self, in_channels, out_channels, kernel_size=3, stride=1, add_norm=True, activation="ReLU", dropout=None):
        """ Module Initializer """
        super().__init__()
        assert activation in ["ReLU", "LeakyReLU", "Sigmoid", "Tanh", None]
        padding = kernel_size // 2
        
        block = []
        block.append(nn.Conv2d(in_channels, out_channels, kernel_size=kernel_size, padding=padding, stride=stride))
        if add_norm:
            block.append(nn.BatchNorm2d(out_channels))
        if activation is not None:
            nonlinearity = getattr(nn, activation, nn.ReLU)()
            if isinstance(nonlinearity, nn.LeakyReLU):
                nonlinearity.negative_slope = 0.2
            block.append(nonlinearity)
            
        if dropout is not None:
            block.append(nn.Dropout(dropout))
            
        self.block =  nn.Sequential(*block)

    def forward(self, x):
        """ Forward pass """
        y = self.block(x)
        return y


class ConvTransposeBlock(nn.Module):
    """
    Simple convolutional block: ConvTranspose + Norm + Act + Dropout
    """
    
    def __init__(self, in_channels, out_channels, kernel_size=3, stride=1, add_norm=True, activation="ReLU", dropout=None):
        """ Module Initializer """
        super().__init__()
        assert activation in ["ReLU", "LeakyReLU", "Tanh", None]
        padding = kernel_size // 2
        
        block = []
        block.append(nn.ConvTranspose2d(in_channels, out_channels, kernel_size=kernel_size, padding=1, stride=stride))
        if add_norm:
            block.append(nn.BatchNorm2d(out_channels))
        if activation is not None:
            nonlinearity = getattr(nn, activation, nn.ReLU)()
            if isinstance(nonlinearity, nn.LeakyReLU):
                nonlinearity.negative_slope = 0.2
            block.append(nonlinearity)
        if dropout is not None:
            block.append(nn.Dropout(dropout))
            
        self.block =  nn.Sequential(*block)

    def forward(self, x):
        """ Forward pass """
        y = self.block(x)
        return y
    

class Reshape(nn.Module):
    """ Reshaping a vector in a given shape """
    
    def __init__(self, shape):
        """ """
        super().__init__()
        self.shape = shape
        
    def forward(self, x):
        """ """
        B, N = x.shape
        x = x.view(B, N, 1, 1)
        y = x.repeat(1, 1, *self.shape)
        return y

# this generator is based on the DCGAN architecture shown in the Session 6 and extended with a label input
# inspiration taken from https://github.com/togheppi/cDCGAN/blob/master/MNIST_cDCGAN_pytorch.py
class Generator(nn.Module):
    """
    A fully convolutional generator using ReLU activations. 
    Takes as input a latent vector and outputs a fake sample.
       (B, latent_dim, 1, 1)  --> (B, num_channels, 32, 32)
    """
    def __init__(self, latent_dim=128, label_dim=10, num_channels=1, base_channels=32):
        """ Model initializer """
        super().__init__()

        layers = []
        latent_layers = []                       # <----------------- this layers get latent vector (gaussian noise) as input.
        label_layers = []                        # <----------------- this layers get onehot-encoded label vector as input. 
        self.output_layer = None
        for i in range(4):
            if i == 0:
                latent_layers.append(
                    ConvTransposeBlock(
                            in_channels=latent_dim,
                            out_channels=base_channels * 2 ** (3-i),
                            kernel_size=4,
                            stride=1,
                            add_norm=True,
                            activation="ReLU"
                        )
                    )
                label_layers.append(
                    ConvTransposeBlock(
                            in_channels=label_dim,
                            out_channels=base_channels * 2 ** (3-i),
                            kernel_size=4,
                            stride=1,
                            add_norm=True,
                            activation="ReLU"
                        )
                    )

            else:
                layers.append(
                    ConvTransposeBlock(
                            in_channels=base_channels * 2 ** (3-i+2),       # <---------------- 2x more channels than in DCGAN from Session 6 (latent_layers and label_layers are concatenated)
                            out_channels=base_channels * 2 ** (3-i+1),      # <---------------- same
                            kernel_size=4,
                            stride=2,
                            add_norm=True,
                            activation="ReLU"
                        )
                    ) 
            
        self.output_layer = (
            ConvTransposeBlock(
                    in_channels=base_channels*2,                            # <---------------- same
                    out_channels=num_channels,
                    kernel_size=4,
                    stride=2,
                    add_norm=False,
                    activation="Tanh"
                )
            )
        
        self.hidden_layers = nn.Sequential(*layers)
        self.latent_layers = nn.Sequential(*latent_layers)
        self.label_layers = nn.Sequential(*label_layers)
        return
    
    def forward(self, x, c):
        """ Forward pass through generator """
        h1 = self.latent_layers(x)
        h2 = self.label_layers(c)
        h = torch.cat([h1, h2], dim=1)              # <----------------- concatenate outputs of latent_layers and label_layers
        h = self.hidden_layers(h)
        y = self.output_layer(h)
        return y

# this discriminator is based on the DCGAN architecture shown in the Session 6 and extended with a label input
# inspiration taken from https://github.com/togheppi/cDCGAN/blob/master/MNIST_cDCGAN_pytorch.py
class Discriminator(nn.Module):
    """ A fully convolutional discriminator using LeakyReLU activations. 
    Takes as input either a real or fake sample and predicts its autenticity.
       (B, num_channels, 32, 32)  -->  (B, 1, 1, 1)
    """
    def __init__(self, in_channels=1, label_dim=10, out_dim=1, base_channels=32, dropout=0.3):
        """ Module initializer """
        super().__init__()  
        
        layers = []
        image_layers = []                       # <----------------- this layers get image as input.
        label_layers = []                       # <----------------- this layers get onehot-encoded label vector as input.
        for i in range(4):
            if i == 0:
                image_layers.append(
                    ConvBlock(
                            in_channels=in_channels,
                            out_channels=base_channels * 2 ** i,
                            kernel_size=4,
                            add_norm=False,
                            activation="LeakyReLU",
                            dropout=dropout,
                            stride=2
                        )
                    )
                label_layers.append(
                    ConvBlock(
                            in_channels=label_dim,
                            out_channels=base_channels * 2 ** i,
                            kernel_size=4,
                            add_norm=False,
                            activation="LeakyReLU",
                            dropout=dropout,
                            stride=2
                        )
                    )
            else:
                layers.append(
                ConvBlock(
                        in_channels=base_channels * 2 if i == 1 else base_channels * 2 ** i,
                        out_channels=base_channels * 2 ** (i + 1),
                        kernel_size=4,
                        add_norm=True,
                        activation="LeakyReLU",
                        dropout=dropout,
                        stride=2
                    )
                )


        self.output_layer = (
                ConvBlock(
                        in_channels=base_channels * 16, 
                        out_channels=out_dim,
                        kernel_size=4,
                        stride=4,
                        add_norm=False,
                        activation="Sigmoid"
                    )
                )
        
        self.hidden_layers = nn.Sequential(*layers)
        self.label_layers = nn.Sequential(*label_layers)
        self.image_layers = nn.Sequential(*image_layers)
        return
      
    def forward(self, x, c):
        """ Forward pass """
        h1 = self.image_layers(x)
        h2 = self.label_layers(c)
        # repeat the label vector to match the image size
        h2 = h2.repeat(1, 1, h1.shape[2], h1.shape[3])
        h = torch.cat([h1, h2], dim=1)              # <----------------- concatenate the two hidden layers
        
        h = self.hidden_layers(h)
        y = self.output_layer(h)
        return y

import torch_fidelity #https://github.com/toshas/torch-fidelity/

class Trainer:
    """
    Class for initializing GAN and training it
    """
    def __init__(self, generator, discriminator, latent_dim=128, writer=None, isConditional=False, training_name="training", use_fidelyty_metrics=False):
        """ Initialzer """
        assert writer is not None, f"Tensorboard writer not set..."

        self.use_fidelyty_metrics = use_fidelyty_metrics
        self.isConditional = isConditional
        self.training_name = training_name
        self.latent_dim = latent_dim
        self.writer = writer 
        self.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.generator = generator.to(self.device)
        self.discriminator = discriminator.to(self.device)
        
        self.optim_discriminator = torch.optim.Adam(self.discriminator.parameters(), lr=3e-4, betas=(0.5, 0.9))
        self.optim_generator = torch.optim.Adam(self.generator.parameters(), lr=3e-4, betas=(0.5, 0.9))
        
        # REAL LABEL = 1
        # FAKE LABEL = 0
        # eps = 1e-10
        # self.criterion_d_real = lambda pred: torch.clip(-torch.log(1 - pred + eps), min=-10).mean()
        # self.criterion_d_fake = lambda pred: torch.clip(-torch.log(pred + eps), min=-10).mean()
        # self.criterion_g = lambda pred: torch.clip(-torch.log(1 - pred + eps), min=-10).mean()
        
        self.criterion_d_real = lambda pred: F.binary_cross_entropy(pred, torch.ones(pred.shape[0], device=pred.device))
        self.criterion_d_fake = lambda pred: F.binary_cross_entropy(pred, torch.zeros(pred.shape[0], device=pred.device))
        self.criterion_g = lambda pred: F.binary_cross_entropy(pred, torch.ones(pred.shape[0], device=pred.device))
        
        self.hist = {
            "d_real": [],
            "d_fake": [],
            "g": []
        }
        return
        
    def train_one_step(self, imgs, labels):
        """ 
        raining both models for one optimization step
        """
        if self.isConditional:
            # One hot encoding of the labels
            labels = F.one_hot(labels, num_classes=10).float()
            labels = labels.view(labels.shape[0], labels.shape[1], 1, 1)
            labels = labels.to(self.device)

        self.generator.train()
        self.discriminator.train()
        
        # Sample from the latent distribution
        B = imgs.shape[0]
        latent = torch.randn(B, self.latent_dim, 1, 1).to(self.device)
        
        # ==== Training Discriminator ====
        self.optim_discriminator.zero_grad()
        # Get discriminator outputs for the real samples
        if self.isConditional:
            prediction_real = self.discriminator(imgs, labels)
        else:
            prediction_real = self.discriminator(imgs)
        # Compute the loss function
        d_loss_real = self.criterion_d_real(prediction_real.view(B))

        # Generating fake samples with the generator
        if self.isConditional:
            fake_samples = self.generator(latent, labels)
        else:
            fake_samples = self.generator(latent)
        # Get discriminator outputs for the fake samples
        if self.isConditional:
            prediction_fake_d = self.discriminator(fake_samples.detach(), labels)
        else:
            prediction_fake_d = self.discriminator(fake_samples.detach()) 
        # Compute the loss function
        d_loss_fake = self.criterion_d_fake(prediction_fake_d.view(B))
        (d_loss_real + d_loss_fake).backward()
        assert fake_samples.shape == imgs.shape
        
        # optimization step
        torch.nn.utils.clip_grad_norm_(self.discriminator.parameters(), 3.0)
        self.optim_discriminator.step()
        
        # === Train the generator ===
        self.optim_generator.zero_grad()
        # Get discriminator outputs for the fake samples
        if self.isConditional:
            prediction_fake_g = self.discriminator(fake_samples, labels)
        else:
            prediction_fake_g = self.discriminator(fake_samples)
        # Compute the loss function
        g_loss = self.criterion_g(prediction_fake_g.view(B))
        g_loss.backward()
        # optimization step
        self.optim_generator.step()
        
        return d_loss_real, d_loss_fake, g_loss
    
    @torch.no_grad()
    def generate(self, N=100):
        """ Generating a bunch of images using current state of generator """
        self.generator.eval()
        latent = torch.randn(N, self.latent_dim, 1, 1).to(self.device)
        if self.isConditional:
            # generate labels
            labels = np.arange(10).repeat(10)
            # convert to torch tensor
            labels = torch.from_numpy(labels).long()
            # one hot encoding
            labels = F.one_hot(labels, num_classes=10).float()
            labels = labels.view(labels.shape[0], labels.shape[1], 1, 1)
            labels = labels.to(self.device)
        
            imgs = self.generator(latent, labels)
        else:
            imgs = self.generator(latent)
        imgs = imgs * 0.5 + 0.5
        return imgs
        
    def train(self, data_loader, N_iters=10000, init_step=0):
        """ Training the models for several iterations """
        
        progress_bar = tqdm(total=N_iters, initial=init_step)
        running_d_loss = 0
        running_g_loss = 0
        
        iter_ = 0
        for i in range(N_iters):
            for real_batch, label_batch in data_loader:           
                real_batch = real_batch.to(self.device)
                d_loss_real, d_loss_fake, g_loss = self.train_one_step(imgs=real_batch, labels=label_batch)
                d_loss = d_loss_real + d_loss_fake
            
                # updating progress bar
                progress_bar.set_description(f"Ep {i+1} Iter {iter_}: D_Loss={round(d_loss.item(),5)}, G_Loss={round(g_loss.item(),5)})")
                
                # adding stuff to tensorboard
                self.writer.add_scalar(f'Loss/Generator Loss', g_loss.item(), global_step=iter_)
                self.writer.add_scalar(f'Loss/Discriminator Loss', d_loss.item(), global_step=iter_)
                self.writer.add_scalars(f'Loss/Discriminator Losses', {
                        "Real Images Loss": d_loss_real.item(),
                        "Fake Images Loss": d_loss_fake.item(),
                    }, global_step=iter_)
                self.writer.add_scalars(f'Comb_Loss/Losses', {
                            'Discriminator': d_loss.item(),
                            'Generator':  g_loss.item()
                        }, iter_)    
                if(iter_ % 200 == 0):
                    imgs = self.generate()
                    grid = torchvision.utils.make_grid(imgs, nrow=10)
                    self.writer.add_image('images', grid, global_step=iter_)
                    torchvision.utils.save_image(grid, os.path.join(os.getcwd(), "imgs", self.training_name + "_training", f"imgs_{iter_}.png"))


                iter_ = iter_ + 1 
                
            if self.use_fidelyty_metrics:
                # inspiration taken from https://github.com/toshas/torch-fidelity/blob/master/examples/sngan_cifar10.py
                metrics = torch_fidelity.calculate_metrics(
                input1=torch_fidelity.GenerativeModelModuleWrapper(self.generator, 128, "normal", 0),
                input1_model_num_samples=20,
                input2="./data/celeba/img_align_celeba_2000", # path to the folder with some images from dataset
                kid_subset_size=20,
                isc=True,
                fid=True,
                kid=True,
                #ppl=True,
                #ppl_epsilon=1e-2,
                #ppl_sample_similarity_resize=64,
                )
        
                # log metrics
                for k, v in metrics.items():
                    self.writer.add_scalar(f'Metrics/{k}', v, global_step=iter_)

            # saving the models
            # TODO add a checkpointing mechanism to save the model if fid score is better than before
                
        return

class CelebGenerator(nn.Module):
    """
    A fully convolutional generator using ReLU activations. 
    Takes as input a latent vector and outputs a fake sample.
       (B, latent_dim, 1, 1)  --> (B, num_channels, 32, 32)
    """
    def __init__(self, latent_dim=256, num_channels=3, base_channels=64):
        """ Model initializer """
        super().__init__()

        layers = []
        for i in range(4):
            layers.append(
                ConvTransposeBlock(
                        in_channels=latent_dim if i == 0 else base_channels * 2 ** (3-i+1),
                        out_channels=base_channels * 2 ** (3-i),
                        kernel_size=4,
                        stride=1 if i == 0 else 2,
                        add_norm=True,
                        activation="ReLU"
                    )
                )

        layers.append(
            ConvTransposeBlock(
                    in_channels=base_channels,
                    out_channels=base_channels,
                    kernel_size=4,
                    stride=2,
                    add_norm=True,
                    activation="ReLU"
                )
            )
        layers.append(
            ConvTransposeBlock(
                    in_channels=base_channels,
                    out_channels=3,
                    kernel_size=4,
                    stride=2,
                    add_norm=False,
                    activation="Tanh"
                )
            )
        
        
        self.model = nn.Sequential(*layers)
        return
    
    def forward(self, x):
        # torch_fidelity sends [B, latent_dim] tensors, so we need to reshape them
        use_fidelity = False
        if len(x.shape) == 2:
            use_fidelity = True
        if use_fidelity:
            x = x.view(x.size(0), x.size(1), 1, 1)

        """ Forward pass through generator """
        y = self.model(x)
        if use_fidelity:
            # y to dtype=torch.uint8 
            y = (y + 1) / 2
            y = y * 255
            y = y.to(dtype=torch.uint8)            
        return y

# based on the code from Session 6 but has 3 input channels instead of 1 and has one more convolution layer

class CelebDiscriminator(nn.Module):
    """ A fully convolutional discriminator using LeakyReLU activations. 
    Takes as input either a real or fake sample and predicts its autenticity.
       (B, num_channels, 32, 32)  -->  (B, 1, 1, 1)
    """
    def __init__(self, in_channels=1, out_dim=1, base_channels=64, dropout=0.3):
        """ Module initializer """
        super().__init__()  
        
        layers = []
        for i in range(4):
            layers.append(
                ConvBlock(
                        in_channels=in_channels if i == 0 else base_channels * 2 ** i,
                        out_channels=base_channels * 2 ** (i + 1),
                        kernel_size=4,
                        add_norm=True,
                        activation="LeakyReLU",
                        dropout=dropout,
                        stride=2
                    )
                )
        layers.append(
                ConvBlock(
                        in_channels=base_channels * 16,
                        out_channels=base_channels * 16,
                        kernel_size=4,
                        stride=4,
                        add_norm=True,
                        activation="LeakyReLU",
                        dropout=dropout,
                    )
                )
        
        layers.append(
                ConvBlock(
                        in_channels=base_channels * 16,
                        out_channels=out_dim,
                        kernel_size=4,
                        stride=4,
                        add_norm=False,
                        activation="Sigmoid"
                    )
                )
        
        self.model = nn.Sequential(*layers)
        return
      
    def forward(self, x):
        """ Forward pass """
        y = self.model(x)
        return y
